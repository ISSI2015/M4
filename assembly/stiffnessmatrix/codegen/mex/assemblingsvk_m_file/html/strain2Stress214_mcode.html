234function stress = strain2Stress(strain,trafo,Mu,Lambda,grad_u0,sigma_u0,weight)
235
236    N = size(strain,1);
237    stress = zeros(size(strain));
238    
239    for i=1:1/2*N
240        stress(2*i-1:2*i,:) = getStressSVK(strain(2*i-1:2*i,:) * (trafo\eye(2)), Mu, Lambda, grad_u0, sigma_u0)' * abs(det(trafo)) * weight;  
241    end
242
243end
